import java.util.Random;

public class MaiorValorMatriz {
    public static void main(String[] args) {
        int[][] matriz = new int[5][5];
        Random random = new Random();

        // preenche a matriz com valores aleatórios não repetidos
        for (int i = 0; i < matriz.length; i++) {
            for (int j = 0; j < matriz[0].length; j++) {
                int num;
                do {
                    num = random.nextInt(25) + 1; // gera um número aleatório entre 1 e 25
                } while (existeNaMatriz(matriz, num)); // repete enquanto o número já existir na matriz
                matriz[i][j] = num;
            }
        }

        // encontra a localização do maior valor na matriz
        int maiorValor = matriz[0][0];
        int linhaMaiorValor = 0;
        int colunaMaiorValor = 0;
        for (int i = 0; i < matriz.length; i++) {
            for (int j = 0; j < matriz[0].length; j++) {
                if (matriz[i][j] > maiorValor) {
                    maiorValor = matriz[i][j];
                    linhaMaiorValor = i;
                    colunaMaiorValor = j;
                }
            }
        }

        // imprime a matriz e a localização do maior valor
        System.out.println("Matriz:");
        for (int i = 0; i < matriz.length; i++) {
            for (int j = 0; j < matriz[0].length; j++) {
                System.out.print(matriz[i][j] + " ");
            }
            System.out.println();
        }
        System.out.println("Maior valor: " + maiorValor);
        System.out.println("Localização: linha " + (linhaMaiorValor + 1) + ", coluna " + (colunaMaiorValor + 1));
    }

    // verifica se um número já existe na matriz
    public static boolean existeNaMatriz(int[][] matriz, int num) {
        for (int i = 0; i < matriz.length; i++) {
            for (int j = 0; j < matriz[0].length; j++) {
                if (matriz[i][j] == num) {
                    return true;
                }
            }
        }
        return false;
    }
}
